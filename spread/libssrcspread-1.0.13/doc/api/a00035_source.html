210  friend bool operator==(const GroupList & groups1,
211  const GroupList & groups2)
212  {
213  unsigned int size = groups1.size();
214 
215  if(size != groups2.size())
216  return false;
217  else
218  while(size-- > 0) {
219  if(std::strncmp(groups1.groups()[size], groups2.groups()[size],

223
